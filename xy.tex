\def\nota#1#2#3{%
  \leavevmode
  \setbox\textbox=\hbox{#3\hskip0pt}%
  \ww=\wd\textbox
  \ifdim \ww<\minw \ww=\minw \fi
  \vbox{%
    \hh=-#1%
    \advance\hh by \max
    \vskip\hh\unit
    \hbox to \ww{%
      \hfil
      \pdfliteral{q #2 rg}%
      \hbox to 0pt{\hss$\bullet$\hss}%
      \pdfliteral{Q}%
      \hbox to 0pt{\hss$\circ$\hss}%
      \hfil
    }%
    \hh=#1%
    \advance\hh by -\min
    \vskip \hh\unit
    %\hbox to \ww{\hfil#3\hfil}%
    \hbox to \ww{%
      \iflead \lead\hfil \else \hfil \fi
      \global\leadfalse
      #3\hfil}%
  }%
  %\ignorespaces
  \hskip\tolskip
  \relax
}
