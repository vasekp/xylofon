\long\def\nota#1#2#3{%
  \leavevmode
  \setbox\textbox=\hbox{#2\hskip0pt}%
  \ww=\wd\textbox
  \ifdim \ww<\minw \ww=\minw \fi
  \underflow=\minw
  \advance\underflow by -\wd\textbox
  \ifdim \underflow<0pt \underflow=0pt \fi
  \divide\underflow by 2
  \ifdim \overflow>\underflow \overflow=\underflow \fi
  \hskip -\overflow
  \vbox{%
    \hh=-#1%
    \advance\hh by \max
    \vskip\hh\unit
    \hbox to \ww{%
      \hfil
      \pdfliteral{q \barva{#1} rg}%
      \hbox to 0pt{\hss$\bullet$\hss}%
      \pdfliteral{Q}%
      \hbox to 0pt{\hss$\circ$\hss}%
      \hfil
    }%
    \hh=#1%
    \advance\hh by -\min
    \vskip \hh\unit
    \hbox to \ww{%
      \iflead \lead\hfil \else \hfil \fi
      #2#3%
      \iflead \lead\hfil \else \hfil \fi
    }%
  }%
  \overflow=0pt
  \relax
}
